exports.getLighterColor = function (rgb, lightness) {
    return {r: (lightness * 255 + (1 - lightness) * rgb.r).toFixed(),
            g: (lightness * 255 + (1 - lightness) * rgb.g).toFixed(),
            b: (lightness * 255 + (1 - lightness) * rgb.b).toFixed()};
};
