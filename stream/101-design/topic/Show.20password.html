@Tim Abbott Are we talking about the following structure for the implementation ?

export function setup_password_visibility(password_field, password_selector) {
    $("body").on("click", password_selector, () => {
        toggle_password_visibility($password_field, password_selector)
    });
}

function toggle_password_visibility($password_field, password_selector){

        if (password_field.attr("type") === "password") {
            password_field.attr("type", "text");
        } else {
            password_field.attr("type", "password");
        }

        $(password_selector).toggleClass("fa-eye fa-eye-slash");
}
