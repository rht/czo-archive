I wasn't able to get your method working, so I just wrote this script that identifies colors in the style-sheets:

import tinycss
from tinycss.color3 import parse_color
def parse_file(filename):
    file = open(filename,"r")
    theme = ''
    stylesheet = tinycss.make_parser().parse_stylesheet(file.read())
    for rule in stylesheet.rules:
        for declaration in rule.declarations:
            for value in declaration.value:
                color = parse_color(value)
                if color is not None:
                    print("Found Color:",color)
                    print("Selector(s):")
                    print(rule.selector.as_css())

def main():
    parse_file("sample.css")

main()
