Ah, we have this code in common.js:

 51 export function adjust_mac_shortcuts(key_elem_class: string, require_cmd_style = false): void {
 52     if (!has_mac_keyboard()) {
 53         return;
 54     }
 55
 56     const keys_map = new Map([
 57         ["Backspace", "Delete"],
 58         ["Enter", "Return"],
 59         ["Home", "Fn + ←"],
 60         ["End", "Fn + →"],
 61         ["PgUp", "Fn + ↑"],
 62         ["PgDn", "Fn + ↓"],
 63         ["Ctrl", "⌘"],
 64     ]);
 65
 66     $(key_elem_class).each(function () {
 67         let key_text = $(this).text();
 68         const keys = key_text.match(/[^\s+]+/g) || [];
 69
 70         if (key_text.includes("Ctrl") && require_cmd_style) {
 71             $(this).addClass("mac-cmd-key");
 72         }
 73
 74         for (const key of keys) {
 75             const replace_key = keys_map.get(key);
 76             if (replace_key !== undefined) {
 77                 key_text = key_text.replace(key, replace_key);
 78             }
 79         }
 80
 81         $(this).text(key_text);
 82     });
 83 }
